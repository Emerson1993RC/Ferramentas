<script>
let chart;

async function lerPDF() {
  const file = document.getElementById('pdfUpload').files[0];
  if (!file) return alert('Selecione um PDF');

  const reader = new FileReader();
  reader.onload = async function() {
    const typedarray = new Uint8Array(this.result);
    const pdf = await pdfjsLib.getDocument(typedarray).promise;

    let textoCompleto = '';

    for (let i = 1; i <= pdf.numPages; i++) {
      const page = await pdf.getPage(i);
      const content = await page.getTextContent();
      const strings = content.items.map(item => item.str);
      textoCompleto += strings.join(' ');
    }

    extrairDados(textoCompleto);
  };

  reader.readAsArrayBuffer(file);
}

function extrairNumero(texto, termo) {
  const regex = new RegExp(termo + "\\s+(\\d+)", "i");
  const match = texto.match(regex);
  return match ? Number(match[1]) : 0;
}

function extrairDados(texto) {
  const agendada = extrairNumero(texto, "Consulta agendada");
  const programada = extrairNumero(texto, "Cuidado continuado");
  const noDia = extrairNumero(texto, "Consulta no dia");
  const urgencia = extrairNumero(texto, "Atendimento de urgência");
  const escuta = extrairNumero(texto, "Escuta inicial");

  document.getElementById('agendada').value = agendada;
  document.getElementById('programada').value = programada;
  document.getElementById('noDia').value = noDia;
  document.getElementById('urgencia').value = urgencia;
  document.getElementById('escuta').value = escuta;

  calcular();
}

function calcular() {
  const agendada = Number(document.getElementById('agendada').value) || 0;
  const programada = Number(document.getElementById('programada').value) || 0;
  const noDia = Number(document.getElementById('noDia').value) || 0;
  const urgencia = Number(document.getElementById('urgencia').value) || 0;
  const escuta = Number(document.getElementById('escuta').value) || 0;

  const programadas = agendada + programada;
  const total = programadas + noDia + urgencia + escuta;

  if (total === 0) return;

  const percentual = (programadas / total) * 100;

  document.getElementById('resultado').innerText = percentual.toFixed(1) + "%";

  let classificacao = "";

  if (percentual <= 30) classificacao = "Crítico";
  else if (percentual <= 50) classificacao = "Regular";
  else if (percentual <= 70) classificacao = "Bom";
  else classificacao = "Ótimo";

  document.getElementById('status').innerText = classificacao;

  gerarGrafico(programadas, total - programadas);
}

function gerarGrafico(programadas, espontaneas) {
  const ctx = document.getElementById('grafico').getContext('2d');

  if (chart) chart.destroy();

  chart = new Chart(ctx, {
    type: 'doughnut',
    data: {
      labels: ['Programada', 'Espontânea'],
      datasets: [{ data: [programadas, espontaneas] }]
    }
  });
}
