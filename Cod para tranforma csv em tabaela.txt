<script>
function getClasse(status) {
  if (status.includes("🔴")) return "red";
  if (status.includes("🟠")) return "orange";
  if (status.includes("🟡")) return "yellow";
  return "";
}

function gerarTabela() {
  const texto = document.getElementById("entrada").value.trim();
  const linhas = texto.split("\n");
  const tbody = document.querySelector("#tabela tbody");
  const cabecalho = document.getElementById("cabecalho");

  tbody.innerHTML = "";
  cabecalho.innerHTML = "";

  // Detecta formato pela quantidade de colunas da primeira linha válida
  const primeiraLinha = linhas.find(l => l.split(",").length >= 6);
  if (!primeiraLinha) return;

  const colunas = primeiraLinha.split(",").length;

  if (colunas === 7) {
    // Formato antigo
    cabecalho.innerHTML = `
      <th>Nome</th>
      <th>Microárea</th>
      <th>Peso/Altura</th>
      <th>PA</th>
      <th>Consulta</th>
      <th>Status</th>
    `;

    linhas.forEach(linha => {
      const col = linha.split(",");
      if (col.length < 7) return;

      const tr = document.createElement("tr");
      tr.className = getClasse(col[6]);

      tr.innerHTML = `
        <td>${col[1]}</td>
        <td>${col[2]}</td>
        <td>${col[3]}</td>
        <td>${col[4]}</td>
        <td>${col[5]}</td>
        <td>${col[6]}</td>
      `;

      tbody.appendChild(tr);
    });

  } else if (colunas === 6) {
    // Novo formato
    cabecalho.innerHTML = `
      <th>Nome</th>
      <th>Microárea</th>
      <th>Status</th>
      <th>PA</th>
      <th>Consulta</th>
    `;

    linhas.forEach(linha => {
      const col = linha.split(",");
      if (col.length < 6) return;

      const tr = document.createElement("tr");
      tr.className = getClasse(col[3]);

      tr.innerHTML = `
        <td>${col[1]}</td>
        <td>${col[2]}</td>
        <td>${col[3]}</td>
        <td>${col[4]}</td>
        <td>${col[5]}</td>
      `;

      tbody.appendChild(tr);
    });
  }
}
</script>
