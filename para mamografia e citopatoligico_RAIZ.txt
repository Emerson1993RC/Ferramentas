    <script>
        // Função para copiar o texto para a área de transferência
        function copyToClipboard(elementId) {
            const textarea = document.getElementById(elementId);
            textarea.select(); // Seleciona o texto dentro do textarea
            textarea.setSelectionRange(0, 99999); // Para dispositivos móveis

            try {
                // Tenta copiar o texto para a área de transferência
                document.execCommand('copy');
                showFeedbackMessage(); // Mostra a mensagem de feedback
            } catch (err) {
                console.error('Erro ao copiar: ', err); // Loga o erro no console
                // Em vez de alert(), você pode mostrar uma mensagem de erro na tela
                // Por exemplo: displayErrorMessage('Não foi possível copiar o texto. Por favor, tente manualmente.');
            }
        }

        // Função para mostrar a mensagem de feedback "Copiado!"
        function showFeedbackMessage() {
            const feedback = document.getElementById('copyFeedback');
            feedback.classList.remove('hidden'); // Torna a mensagem visível
            feedback.classList.add('opacity-100'); // Garante opacidade total

            // Esconde a mensagem após 2 segundos
            setTimeout(() => {
                feedback.classList.remove('opacity-100'); // Inicia a transição para esconder
                feedback.classList.add('opacity-0'); // Torna a mensagem transparente
                // Após a transição, esconde completamente a mensagem para não ocupar espaço
                setTimeout(() => {
                    feedback.classList.add('hidden');
                }, 300); // Tempo da transição CSS (deve corresponder ao transition-duration)
            }, 2000);
        }
    </script>
